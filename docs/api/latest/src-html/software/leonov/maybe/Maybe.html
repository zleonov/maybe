018import static java.util.Objects.requireNonNull;
019
020import java.nio.file.Files;
021import java.nio.file.LinkOption;
022import java.nio.file.Path;
023import java.util.Map;
024import java.util.NoSuchElementException;
025import java.util.Objects;
026import java.util.Optional;
027import java.util.function.Consumer;
028import java.util.function.Function;
029import java.util.function.Predicate;
030import java.util.function.Supplier;
031import java.util.stream.Stream;

033/**
034 * A immutable container which may or may not hold a {@code nullable} value. This class is the analog of Java's
035 * {@link Optional} class with the capability to differentiate between {@code null} values and <i>absent</i> values.
036 * <p>
037 * This is a <a href="https://docs.oracle.com/javase/8/docs/api/java/lang/doc-files/ValueBased.html" target=
038 * "_blank">value-based</a> class; use of identity-sensitive operations (including reference equality ({@code ==}),
039 * identity hash code, or synchronization) on instances of this class may have unpredictable results and should be
040 * avoided.
041 *
042 * @apiNote The purpose of this class is to allow users to differentiate between a {@code null} value and an
043 *          {@link #isPresent() absent} or unspecified value. It is gently encouraged to use this class as a method
044 *          return type rather than a parameter. In either case a reference to this class <b>should never itself</b> be
045 *          {@code null}.
046 * 
047 * @author Zhenya Leonov
048 */
049public final class Maybe<T> {
050
051    private static final Maybe<?> ABSENT = new Maybe<>();
052
053    private final T       value;
054    private final boolean isPresent;
055
056    private volatile Optional<T> optional;
057
058    private Maybe() {
059        value     = null;
060        isPresent = false;
061    }
062
063    private Maybe(final T value) {
064        this.value = value;
065        isPresent  = true;
066    }
067
068    /**
069     * Returns a {@link Maybe} instance which contains no value.
070     * 
071     * @return a {@link Maybe} instance which contains no value
072     */
073    @SuppressWarnings("unchecked")
074    public static <T> Maybe<T> absent() {
075        return (Maybe<T>) ABSENT;
076    }
077
078    /**
079     * Returns a {@link Maybe} instance containing the result of {@link Map#get(Object) map.get(key)} if such a mapping
080     * exists or an {@link #absent() absent} instance otherwise.
081     * <p>
082     * <b>Discussion:</b> For maps which permit {@code null} values the {@link Map#get(Object)} method returns a
083     * {@code null} both when a key is not found in the map, or if the key is explicitly mapped to {@code null}, requiring
084     * users to call {@link Map#containsKey(Object) containsKey(Object)} to distinguish between the two cases. This has
085     * historically been an overwhelmingly common source of errors.
086     * <p>
087     * This method is intended to be used as a safer a replacement which always produces deterministic results.
088     * <p>
089     * For example:
090     * 
091     * <pre>{@code
092     * if (map.containsKey(key)) {
093     *     final Object value = map.get(key);
094     *     System.out.println(value);
095     * }
096     * }</pre>
097     * 
098     * Can be rewritten as:
099     * 
100     * <pre>{@link
101     * Maybe}.get(map, object).{@link #ifPresent(Consumer) ifPresent(System.out::println)};
102     * }</pre>
103     * 
104     * @param <V> the type of values stored in the specified map
105     * @param map the specified map
106     * @param key the key whose associated value is to be returned
107     * @return a {@link Maybe} instance containing the result of {@link Map#get(Object) Map.get(key)} if such a mapping
108     *         exists or an {@link #absent() absent} instance otherwise
109     * @throws ClassCastException   if the {@code key} is of an inappropriate class type for this map
110     * @throws NullPointerException if the {@code key} is {@code null} and this map does not permit {@code null} keys
111     */
112    public static <V> Maybe<V> get(final Map<?, ? extends V> map, final Object key) {
113        requireNonNull(map, "map == null");
114        return map.containsKey(key) ? Maybe.of(map.get(key)) : Maybe.absent();
115    }
116
117    /**
118     * Returns a {@link Maybe} instance which contains the specified possibly {@link #isNull null} value.
119     * 
120     * @param value the specified value
121     * @return a {@link Maybe} instance which contains the specified possibly {@link #isNull null} value
122     */
123    public static <T> Maybe<T> of(final T value) {
124        return new Maybe<>(value);
125    }
126
127    /**
128     * Returns a {@link Maybe} instance which contains the specified {@code path} if it
129     * {@link Files#exists(Path, LinkOption...) exists} or an {@link #absent() absent} instance otherwise.
130     * 
131     * @param path    the specified path
132     * @param options indicating how symbolic links are handled
133     * @return a {@link Maybe} instance which contains the specified {@code path} if it
134     *         {@link Files#exists(Path, LinkOption...) exists} or an {@link #absent() absent} instance otherwise
135     */
136    public static Maybe<Path> get(final Path path, final LinkOption... options) {
137        requireNonNull(path, "path == null");
138        requireNonNull(options, "options == null");
139        return Files.exists(path, options) ? Maybe.of(path) : Maybe.absent();
140    }
141
142    /**
143     * Returns the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or the {@code defaultValue}
144     * otherwise.
145     * 
146     * @param defaultValue the default value to return if no value is {@link #isPresent() present}
147     * @return the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or the {@code defaultValue}
148     *         otherwise
149     */
150    public T orElse(final T defaultValue) {
151        return isPresent ? value : defaultValue;
152    }
153
154    /**
155     * Returns {@code this} {@link Maybe} instance if the value is {@link #isPresent() present} and satisfies the given
156     * {@code Predicate} or an {@link #absent() absent} instance otherwise.
157     *
158     * @param predicate the predicate to apply to the value if is present
159     * @return {@code this} {@link Maybe} instnace if the value is {@link #isPresent() present} and satisfies the given
160     *         {@code Predicate} or an {@link #absent() absent} instance otherwise
161     */
162    public Maybe<T> filter(final Predicate<? super T> predicate) {
163        requireNonNull(predicate, "predicate == null");
164        if (!isPresent)
165            return this;
166        else
167            return predicate.test(value) ? this : absent();
168    }
169
170    /**
171     * Returns the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or throws a
172     * {@code NoSuchElementException} otherwise.
173     * 
174     * @return the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or throws a
175     *         {@code NoSuchElementException} otherwise
176     * @throws NoSuchElementException if no value is {@link #isPresent() present}
177     */
178    public T get() {
179        if (isPresent)
180            return value;
181        throw new NoSuchElementException();
182    }
183
184    /**
185     * {@link Consumer#accept(Object) Invokes} the specified {@code Consumer} if the value is {@link #isPresent() present}
186     * and not {@link #isNull() null}.
187     * 
188     * @param consumer the specified consumer
189     * @return {@code this} {@link Maybe} instance
190     */
191    public Maybe<T> ifNotNull(final Consumer<? super T> consumer) {
192        requireNonNull(consumer, "consumer == null");
193        if (value != null)
194            consumer.accept(value);
195        return this;
196    }
197
198    /**
199     * Returns {@code this} {@link Maybe} instance if the value is not {@link #isNull() null} or the {@code other} instance
200     * otherwise.
201     * 
202     * @param other the {@link Maybe} instance to return if the value is {@link #isNull() isNull}
203     * @return {@code this} {@link Maybe} instance if the value is not {@link #isNull() null} or the {@code other} instance
204     *         otherwise
205     */
206    @SuppressWarnings("unchecked")
207    public Maybe<T> ifNull(final Maybe<? extends T> other) {
208        requireNonNull(other, "other == null");
209        return isNull() ? (Maybe<T>) other : this;
210    }
211
212    /**
213     * {@link Runnable#run() Invokes} the specified {@code Runnable} if the value is is {@link #isNull() null}.
214     * 
215     * @param runnable the specified runnable
216     * @return {@code this} {@link Maybe} instance
217     */
218    public Maybe<T> ifNull(final Runnable runnable) {
219        requireNonNull(runnable, "runnable == null");
220        if (isNull())
221            runnable.run();
222        return this;
223    }
224
225    /**
226     * Throws an exception produced by the specified {@code Supplier} if the value is {@link #isNull() null} or returns
227     * {@code this} {@link Maybe} instance otherwise.
228     * 
229     * @param <X>      the type of exception to be thrown
230     * @param supplier the exception supplier
231     * @return this {@link Maybe} instance if the value is not {@link #isPresent present} or is not {@link #isNull() null}
232     * @throws X if the value is {@link #isNull() null}
233     */
234    public <X extends Throwable> Maybe<T> ifNullThrow(final Supplier<? extends X> supplier) throws X {
235        requireNonNull(supplier, "supplier == null");
236        if (isNull())
237            throw requireNonNull(supplier.get(), "Supplier.get() == null");
238        return this;
239    }
240
241    /**
242     * {@link Consumer#accept(Object) Invokes} the specified {@code Consumer} if the value is {@link #isPresent() present}.
243     * 
244     * @param consumer the specified consumer
245     * @return {@code this} {@link Maybe} instance
246     */
247    public Maybe<T> ifPresent(final Consumer<? super T> consumer) {
248        requireNonNull(consumer, "consumer == null");
249        if (isPresent)
250            consumer.accept(value);
251        return this;
252    }
253
254    /**
255     * Returns {@code true} if the value is {@link #isPresent() present} and is {@code null} or {@code false} otherwise.
256     * 
257     * @return {@code true} if the value is {@link #isPresent() present} and is {@code null} or {@code false} otherwise
258     */
259    public boolean isNull() {
260        return isPresent && value == null;
261    }
262
263    /**
264     * Returns {@code true} if the value is present or {@code false} otherwise.
265     * 
266     * @return {@code true} if the value is present or {@code false} otherwise
267     */
268    public boolean isPresent() {
269        return isPresent;
270    }
271
272    /**
273     * Returns a {@link Maybe} instance which contains the result of applying the given {@code Function} to the current
274     * value if it is {@link #isPresent() present} or an {@link #absent() absent} instance otherwise.
275     *
276     * 
277     * @param function the mapping function to apply to the value if it is {@link #isPresent() present}
278     * @return a {@link Maybe} instance which contains the result of applying the given {@code Function} to the current
279     *         value if it is {@link #isPresent() present} or an {@link #absent() absent} instance otherwise
280     */
281    public <U> Maybe<U> map(final Function<? super T, ? extends U> function) {
282        requireNonNull(function, "function == null");
283        return isPresent ? Maybe.of(function.apply(value)) : absent();
284    }
285
286    /**
287     * Returns {@code this} {@link Maybe} instance if the value is {@link #isPresent() present} or the {@code other}
288     * instance otherwise.
289     * 
290     * @param other the {@link Maybe} instance to return if the value is not {@link #isPresent() present}
291     * @return {@code this} {@link Maybe} instance if the value is {@link #isPresent() present} or the {@code other}
292     *         instance otherwise
293     */
294    @SuppressWarnings("unchecked")
295    public Maybe<T> or(final Maybe<? extends T> other) {
296        requireNonNull(other, "other == null");
297        return isPresent ? this : (Maybe<T>) other;
298    }
299
300    /**
301     * {@link Runnable#run() Invokes} the specified {@code Runnable} if the value is not {@link #isPresent() present}.
302     * 
303     * @param runnable the specified runnable
304     * @return {@code this} {@link Maybe} instance
305     */
306    public Maybe<T> orElse(final Runnable runnable) {
307        requireNonNull(runnable, "runnable == null");
308        if (!isPresent)
309            runnable.run();
310        return this;
311    }
312
313    /**
314     * Returns the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or {@link Supplier#get()}
315     * otherwise.
316     * 
317     * @param supplier the {@code Supplier} whose result is returned if no value is {@link #isPresent() present}
318     * @return the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or {@link Supplier#get()}
319     *         otherwise
320     */
321    public T orElse(final Supplier<? extends T> supplier) {
322        requireNonNull(supplier, "supplier == null");
323        return isPresent ? value : supplier.get();
324    }
325
326    /**
327     * Returns this {@link Maybe} instance if the value is {@link #isPresent() present} or throws an exception produced by
328     * the specified {@link Supplier} otherwise.
329     * 
330     * @param <X>      the type of exception to be thrown
331     * @param supplier the exception supplier
332     * @return this {@link Maybe} instance if the value is {@link #isPresent() present}
333     * @throws X if the value is not {@link #isPresent() present}
334     */
335    public <X extends Throwable> Maybe<T> orElseThrow(final Supplier<? extends X> supplier) throws X {
336        requireNonNull(supplier, "supplier == null");
337        if (isPresent)
338            return this;
339        else
340            throw requireNonNull(supplier.get(), "Supplier.get() == null");
341    }
342
343    /**
344     * Returns the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or the {@code null}
345     * otherwise.
346     * 
347     * @return the possibly {@link #isNull() null} value if it is {@link #isPresent() present} or the {@code null} otherwise
348     */
349    public T orNull() {
350        return orElse((T) null);
351    }
352
353    /**
354     * Returns a sequential {@link Stream} containing the value if it is {@link isPresent present} or an
355     * {@link Stream#empty() empty} {@code Stream} otherwise.
356     *
357     * @return a sequential {@link Stream} containing the value if it is {@link isPresent present} or an
358     *         {@link Stream#empty() empty} {@code Stream} otherwise
359     */
360    public Stream<T> stream() {
361        return isPresent ? Stream.of(value) : Stream.empty();
362    }
363
364    /**
365     * Returns this {@link Maybe} as a Java {@link Optional}.
366     * <p>
367     * <b>Note:</b> A {@link #isNull() null} value will be treated as an {@link Optional#ofNullable(Object) empty} value by
368     * Java's {@code Optional}.
369     * 
370     * @implNote This method uses <a href="https://en.wikipedia.org/wiki/Double-checked_locking" target="_blank">
371     *           Double-Checked Locking</a> to lazily return a single {@code Optional} instance and is safe to use in a
372     *           concurrent environment.
373     * 
374     * @return {@code this} {@link Maybe} as a Java {@link Optional}
375     */
376    public Optional<T> toOptional() {
377        Optional<T> ref = optional;
378        if (ref == null)
379            synchronized (this) {
380                ref = optional;
381                if (ref == null)
382                    optional = ref = Optional.ofNullable(value);
383            }
384        return ref;
385    }
386
387    @Override
388    public boolean equals(Object obj) {
389        if (this == obj)
390            return true;
391
392        if (obj == null || getClass() != obj.getClass())
393            return false;
394
395        final Maybe<?> other = (Maybe<?>) obj;
396
397        return isPresent == other.isPresent && Objects.equals(value, other.value);
398    }
399
400    @Override
401    public int hashCode() {
402        return Objects.hash(isPresent, value);
403    }
404
405    @Override
406    public String toString() {
407        return String.format("%s[%s", Maybe.class.getSimpleName(), isPresent() ? value + "]" : "]");
408    }
